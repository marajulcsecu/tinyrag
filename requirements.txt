# ----------------------------------------------------------------------------

# ----------------------------------------------------------------------------
pyyaml==6.0.2                   # config.yaml loader

pydantic-settings==2.6.1        # settings-as-a-class pattern for config

# ----------------------------------------------------------------------------
# 3. Vector store & embeddings
# ----------------------------------------------------------------------------
faiss-cpu==1.8.0.post1          # CPU-only FAISS — no GPU dep
sentence-transformers==3.2.1    # embedding model wrapper
# NOTE on torch: PyPI's default `torch==2.4.1` is the CUDA build (~2 GB).
# To get the CPU-only wheel (~250 MB), install with the PyTorch CPU index:
#   pip install --index-url https://download.pytorch.org/whl/cpu torch==2.4.1
# `setup.sh` does this automatically. If you pip install -r requirements.txt
# directly on a CPU-only machine (Pi, or laptop without NVIDIA GPU), the
# CUDA build installs harmlessly but wastes disk — runtime is identical
# because torch.cuda.is_available() returns False either way.
torch==2.4.1
numpy==1.26.4                   # pinned for faiss-cpu 1.8.x compatibility
tiktoken==0.8.0                 # token counting for chunking & prompts


# ----------------------------------------------------------------------------
# 4. Document parsing
# ----------------------------------------------------------------------------
pdfplumber==0.11.4              # robust PDF text extraction


# ----------------------------------------------------------------------------
# 5. HTTP client (talks to llama.cpp HTTP server)
# ----------------------------------------------------------------------------
httpx==0.27.2                   # async + sync HTTP client
sse-starlette==2.1.3            # Server-Sent Events for streaming LLM tokens


# ----------------------------------------------------------------------------
# 6. Structured logging
# ----------------------------------------------------------------------------
structlog==24.4.0               # JSON-native logging


# ----------------------------------------------------------------------------
# 7. Sensor I/O — laptop (no GPIO)

# ----------------------------------------------------------------------------
# 8. Data handling
# ----------------------------------------------------------------------------
